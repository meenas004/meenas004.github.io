function BootstrapTooltip(props) {

/**
 * KPI Card component \n
 * The component needs following attributes: \n
 * 1. Heading 2. Value 3. Tooltip text 4. Color
 */

class ValueBox extends React.Component{
    constructor(props){
        super(props)
    }
    render(){
        // const classes = useStyles();
        return(
            <React.Fragment>
            {(this.props.data.tooltip && this.props.data.tooltip !== "") && <BootstrapTooltip title = {ReactHtmlParser(this.props.data.tooltip)} placement="top">
            <Card  style={{paddingBottom: "10px", marginLeft: '10px', marginRight: '10px', background: this.props.color?this.props.color:"white"}}>
                <CardContent>
                    <Typography gutterBottom variant="h5" component="h2" style={{color:this.props.color?this.props.color.title:"black"}}> {ReactHtmlParser(this.props.data.heading)} </Typography>
                    <Typography variant = "h4" component = "h1" color={this.props.color?this.props.color.value:"black"}>{ReactHtmlParser(this.props.data.value)}</Typography>
                    <Typography variant="body2" style={{color:this.props.color?this.props.color.subtext:"#E5E5E5"}}  component="p">{ReactHtmlParser(this.props.data.subtext)}</Typography>
                </CardContent>
            </Card>
            </BootstrapTooltip>}
            {(!this.props.data.tooltip || this.props.data.tooltip === "") && 
            <Card  style={{paddingBottom: "10px", marginLeft: '10px', marginRight: '10px', background: this.props.color?this.props.color:"white"}}>
                <CardContent>
                    <Typography gutterBottom variant="h5" component="h2" style={{color:this.props.color?this.props.color.title:"black"}}> {ReactHtmlParser(this.props.data.heading)} </Typography>
                    <Typography variant = "h4" component = "h1" color={this.props.color?this.props.color.value:"black"}>{ReactHtmlParser(this.props.data.value)}</Typography>
                    <Typography variant="body2" style={{color:this.props.color?this.props.color.subtext:"#E5E5E5"}}  component="p">{ReactHtmlParser(this.props.data.subtext)}</Typography>
                </CardContent>
            </Card>
            }
            </React.Fragment>
        )
    }
}

export default ValueBox;
