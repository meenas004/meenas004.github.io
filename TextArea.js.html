import React from 'react';
import TextareaAutosize from '@material-ui/core/TextareaAutosize';

/**
 * Implementation of TextArea using material-ui TextAutoSize component. TextArea component is used to take inputs for paragraphs. 
 * @component
 * @property {string} value State component element to keep track of the value put in by user
 */
class TextArea extends React.Component {
    constructor(props){
        super(props)
        this.state={
            value: this.props.selectedOption
        }
    }
    /**
     * 
     * @property {json} e event parameter passsed by component in case of change event 
     */
    handleChange = (e)  => {
        
        this.setState(
            {
                value: e.target.value
            })
    }

    render(){
        return(
            <TextareaAutosize
                style={{width: '380px'}}
                disabled = {this.props.disabled}
                onChange = {this.handleChange}
                onBlur = {() => {this.props.handleChange(this.state.value, this.props.pos, this.props.parent_pos, this.props.el_pos, this.props.id)}}
                rowsMin = {5}
                aria-label="maximum height"
                value={this.state.value}
                placeholder={this.props.label} />
        )
    }
}

export default TextArea;
