import React from 'react';
import {Radio, RadioGroup, FormControl, FormControlLabel, FormLabel, withStyles, makeStyles} from '@material-ui/core';
import { RadioButton } from '@material-ui/core';

/**
 * Implementation of RadioButtonGroup using RadioGroup element of material-ui
 * @component
 * @property {json} value object to track selected option 
 */

class RadioButtonsGroup extends React.Component {
    constructor(props){
      super(props)
      this.state = {
          value: this.props.selectedOption
      }
    }
    /**
     * Handler to handle input change. The function sets value (state variable) equal to selected option. The function converts the value supplied by component into json object 
     * @property {json} event 
     * 
     */
    handleChange = (event) => {
     this.setState({
        value: event.target.value
     })
     this.props.handleChange(event.target.value, this.props.pos, this.props.parent_pos, this.props.el_pos, this.props.id);
    };

    componentWillMount(){
      this.setState({
        value: this.props.selectedOption
      })
    }

  componentWillReceiveProps(next) {
    if(next.options.join('') === this.props.options.join('')) return null;
    this.setState({
        valule: this.props.selectedOption
    })
  }
   render(){ 
    var options = this.props.options; 
    return (
      <FormControl component="fieldset">
        <FormLabel component="legend">{this.props.label}</FormLabel>
        <RadioGroup row aria-label="gender" name="gender1" value={this.state.value} onChange={this.handleChange} >
          {
              options.map(obj=> {
                  return(
                      <FormControlLabel value = {obj.value} control = {<Radio  />} label = {obj.label} />
                  )
              })
          }
        </RadioGroup>
      </FormControl>
  );}
}

export default RadioButtonsGroup;
