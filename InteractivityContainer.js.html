import {SelectInput, RadioButtonGroup} from './../../components/Interactivity'
import React from 'react'
/**
 * Component to display interactivities
 * @param {props} props props passed by ChartContainer
 */
const InteractivityContainer = (props) => {
    var isRadio = props.isRadio;
    return(
        <div>
        {isRadio && <RadioButtonGroup pos = {props.pos} options= {props.options} handleChange = {props.handleChange} label = {props.label}/> }
        {!isRadio && <SelectInput pos={props.pos} options= {props.options} handleChange = {props.handleChange} label = {props.label}/>}
        </div>
    )
}

export default InteractivityContainer
