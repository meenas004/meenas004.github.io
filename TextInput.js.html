import { TextField } from '@material-ui/core';
import React from 'react';


/**
 * Implementation of Text Input using material-ui TextField component 
 * @component
 *  @property {json} value Value to track value input by user
 */
class TextInput extends React.Component{
    constructor(props){
        super(props);
        this.state = {
            value: this.props.selectedOption
        }
    }

    /** 
     * @property {json} e event parameter passsed by component in case of change event 
     */

    handleChange = (e) => {
        this.setState(
            {
                value: e.target.value
            })
    }
    /**
     * Remove selected option whenever it makes sense
     * @param {json} next 
     */
    componentWillReceiveProps(next){
        if(next.label === this.props.label) return null;
        this.setState({
            value: this.props.selectedOption
        })
    }

    render(){
        return(
            <TextField value = {this.state.value} onBlur={() => {this.props.handleChange(this.state.value, this.props.pos, this.props.parent_pos, this.props.el_pos, this.props.id)}} onChange={this.handleChange} label = {this.props.label} />
        )
    }
}

export default TextInput
